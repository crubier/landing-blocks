# Pricing

        prices={[
            {
                heading: 'FREE',
                subheading: 'Essential Features',
                features: ['Unlimited', '1200', '2Gb', '2 Weeks'],
                price: 'Free',
                cta: <Link color='blue.600'>Get started</Link>,
            },
            {
                heading: 'PLUS',
                subheading: 'Essential Features',
                features: ['Unlimited', '1200', '2Gb', '2 Weeks', 'x'],
                price: '$10',
                background: '#D1F0FD',
                cta: <Link color='blue.600'>Get started</Link>,
            },
            {
                heading: 'PRO',
                subheading: 'Essential Features',
                features: ['Unlimited', '1200', '2Gb', '2 Weeks', 'x', 'x'],
                price: '$20',
                background: '#FFEAB6',
                cta: <Link color='blue.600'>Get started</Link>,
